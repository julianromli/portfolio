# Portfolio Migration: Static HTML → Next.js 16 + Base UI

## Summary

Convert static HTML/CSS/JS portfolio to Next.js 16 with App Router, Base UI components, and Tailwind CSS. Multi-page routing with modernized dark theme.

## Tech Stack

| Layer | Technology |
|-------|------------|
| Framework | Next.js 16 (App Router, TypeScript, Turbopack) |
| UI Components | Base UI (@base-ui/react) |
| Styling | Tailwind CSS v4 |
| Icons | Lucide React (replacing Ionicons) |
| Fonts | Google Fonts - Poppins (next/font) |
| Forms | Server Actions |

## Routes

```
/                 → Home/About (default)
/resume           → Resume (Education, Experience, Skills)
/portfolio        → Portfolio grid with category filter
/blog             → Blog posts grid
/contact          → Contact form + info
```

### 3.2 Navigation with Base UI

```tsx
// components/layout/Navigation.tsx
'use client'
import Link from 'next/link'
import { usePathname } from 'next/navigation'
import { NavigationMenu } from '@base-ui/react/navigation-menu'

const links = [
  { href: '/', label: 'About' },
  { href: '/resume', label: 'Resume' },
  { href: '/portfolio', label: 'Portfolio' },
  { href: '/blog', label: 'Blog' },
  { href: '/contact', label: 'Contact' },
]

export function Navigation() {
  const pathname = usePathname()
  
  return (
    <NavigationMenu.Root>
      <NavigationMenu.List className="flex gap-4">
        {links.map(({ href, label }) => (
          <NavigationMenu.Item key={href}>
            <NavigationMenu.Link
              render={<Link href={href} />}
              className={`px-4 py-2 rounded transition-colors
                ${pathname === href 
                  ? 'text-accent bg-surface-1' 
                  : 'text-foreground-muted hover:text-accent'}`}
            >
              {label}
            </NavigationMenu.Link>
          </NavigationMenu.Item>
        ))}
      </NavigationMenu.List>
    </NavigationMenu.Root>
  )
}
```

## Phase 6: Contact Form Server Action

```tsx
// app/contact/actions.ts
'use server'

import { z } from 'zod'

const schema = z.object({
  name: z.string().min(2),
  email: z.string().email(),
  message: z.string().min(10),
})

export async function submitContact(formData: FormData) {
  const data = {
    name: formData.get('name'),
    email: formData.get('email'),
    message: formData.get('message'),
  }
  
  const result = schema.safeParse(data)
  if (!result.success) {
    return { error: result.error.flatten() }
  }
  
  // TODO: Send email or store in database
  console.log('Contact submission:', result.data)
  
  return { success: true }
}
```
